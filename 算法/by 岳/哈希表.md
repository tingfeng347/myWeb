public class Solution {
    public boolean hasCycle(ListNode head) {
      Set<ListNode> s=new HashSet<ListNode>();
      while(head!=null){
          if(!s.add(head)){//表示没有添加这个头节点，因为不允许重复，链表里已经有一个了。
              return true;
          }
          head=head.next;
      }
        return false;
    }
}
